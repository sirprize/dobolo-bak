Affix

Add affix behavior to any element. Use offsets to define when to toggle the pinning of an element on and off.

Declarative

<div data-dojo-type="dobolo/Affix" data-dojo-props="offsetTop: 100">
    ...
</div>

Programmatic

require(["dobolo/Affix", "dojo/dom"], function (Affix, dom) {
    var affix = new Affix({
        offsetTop: 100,
        offsetBottom: 100
    }, dom.byId('my-affix'));
});

Properties

Name Type Default
offsetTop Number 0
offsetBottom Number 0

Button

Stateful Button

Declarative

<script>
    var load = function (ev) {
        var button = this;
        button.loading();
        setTimeout(function () { button.reset(); }, 2000);
    }
</script>

<button class="btn btn-primary" data-dojo-type="dobolo/Button" data-dojo-props="onClick: load, loadingText: 'Loading...', resetText: 'Loaded'">Load</button>

Programmatic

require(["dobolo/Button"], function (Button) {
    var button = new Button({
        type: 'submit',
        'class': 'btn btn-primary',
        label: 'Load',
        loadingText: 'Loading...',
        resetText: 'Loaded'
    }, 'my-button');

    button.startup();
    
    button.on('click', function (ev) {
        button.loading();
        setTimeout(function () {
            button.reset();
        }, 2000);
    });
});

Properties

Name Type Default
loadingText String Loading...
resetText String Loaded

Methods

Name Args Returns
loading() -- undefined
reset() -- undefined
